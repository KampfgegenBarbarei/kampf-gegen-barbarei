# Deutsch-Portugal Events

add_namespace = Portugal

country_event = { #Colonial Instability
	id = Portugal.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event Portugal.1 FROM = [From.GetName]" }
	title = Portugal.1.t
	desc = Portugal.1.d
	picture = GFX_report_event_generic_towar

	fire_only_once = yes
	
	trigger = {
		tag = POR
		date > 1953.02.02

	}

	option = { #Surely things will solve themselves
		name = Portugal.1.a
		hidden_effect = {
			set_country_flag = POR_colonial_instability
		}
		add_stability = -0.05
	}
}

country_event = { #Angola War
	id = Portugal.2
	title = Portugal.2.t
	desc = Portugal.2.d
	picture = GFX_report_event_generic_towar

	is_triggered_only = yes

	option = { #We'll crush them soon.
		name = Portugal.2.a
		hidden_effect = {
			set_global_flag = POR_angola_war
			

			ANG = {
				add_manpower = 50000

				#states
				transfer_state = 540


				POR = {
					declare_war_on = {
						target = ANG
						type = annex_everything
					}
				}
				
				add_equipment_to_stockpile = {
					type = infantry_equipment_1
					amount = 5000
					producer = CSA
				}
				add_equipment_to_stockpile = {
					type = artillery_equipment_1
					amount = 150
					producer = CSA
				}
				
				load_oob = "ANG"
			}
			POR = {
				load_oob = "POR_angola"
			}
			news_event = {
				id = worldnews.7
				days = 1
			}
		}
		add_manpower = -10000
	}
}
